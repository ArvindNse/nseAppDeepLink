  <script>

    const iosScheme = 'nseindia://';

    const androidScheme = 'intent://open-app#Intent;scheme=nseindia;package=com.nse.nseindia;end';


    const appStoreLink = ‘https://apps.apple.com/in/app/nseindia/id6736678487’;

    const playStoreLink = 'https://play.google.com/store/apps/details?id=com.nse.nseindia';


    function openApp() {

      const userAgent = navigator.userAgent || navigator.vendor || window.opera;


      if (/android/i.test(userAgent)) {

        window.location = androidScheme;

        setTimeout(() => {

          window.location = playStoreLink;

        }, 2000);

      } else if (/iPad|iPhone|iPod/.test(userAgent) && !window.MSStream) {

        window.location = iosScheme;

        setTimeout(() => {

          window.location = appStoreLink;

        }, 2000);

      } else {

        window.location = playStoreLink;

      }

    }


    window.onload = openApp;
